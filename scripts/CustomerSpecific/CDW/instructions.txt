# Instructions for Building a Flask Video Processing Web Application

## Objective
Develop a Flask-based web application that allows users to upload a basketball game video, processes it using a machine learning model to identify and extract clips showing players on the court (excluding commercials, fan close-ups, or other non-gameplay footage), and displays the video with a progress bar and thumbnails for each extracted clip. The application must be user-friendly, visually appealing, and optimized for usability.

## Prerequisites
- **Python Environment**: Use Python 3.8+ with dependencies: Flask, OpenCV (`opencv-python`), NumPy, and a machine learning model framework (e.g., TensorFlow or PyTorch) compatible with the provided model.
- **Model**: A pre-trained machine learning model capable of classifying video frames or segments into multiple classes, including at least one class representing "players on court" (e.g., "gameplay").
- **Frontend**: Use HTML, JavaScript, and Tailwind CSS (via CDN) for styling.
- **Video Processing**: Use OpenCV for video frame extraction and clip generation.
- **Storage**: Ensure temporary storage for uploaded videos and generated clips on the server.

import os
import cv2
import numpy as np
import json
from flask import Flask, request, render_template, jsonify, send_file
from werkzeug.utils import secure_filename
import tensorflow as tf  # Adjust for PyTorch if needed

app = Flask(__name__)

# Configuration
UPLOAD_FOLDER = 'static/uploads'
ALLOWED_EXTENSIONS = {'mp4', 'avi', 'mov'}
app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER
os.makedirs(UPLOAD_FOLDER, exist_ok=True)

# Load model and config
with open('config.json', 'r') as f:
    config = json.load(f)
MODEL = tf.keras.models.load_model('model/model_file.h5')  # Adjust for PyTorch
TARGET_CLASS = config['target_class']

def allowed_file(filename):
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

def preprocess_frame(frame):
    # Resize frame to model input size (e.g., 224x224)
    frame = cv2.resize(frame, (224, 224))
    frame = frame / 255.0  # Normalize
    frame = np.expand_dims(frame, axis=0)  # Add batch dimension
    return frame

def process_video(video_path):
    cap = cv2.VideoCapture(video_path)
    fps = cap.get(cv2.CAP_PROP_FPS)
    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    clips = []
    current_clip = []
    clip_id = 0
    thumbnails = []
    frame_count = 0

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break

        # Preprocess frame and predict
        processed_frame = preprocess_frame(frame)
        prediction = MODEL.predict(processed_frame)
        predicted_class = config['classes'][np.argmax(prediction)]

        # If gameplay, add to current clip
        if predicted_class == TARGET_CLASS:
            if not current_clip:  # Start new clip
                clip_start = frame_count / fps
            current_clip.append(frame)
        else:
            if current_clip:  # End current clip
                if len(current_clip) > fps:  # Minimum 1-second clip
                    clip_path = os.path.join(app.config['UPLOAD_FOLDER'], f'clip_{clip_id}.mp4')
                    thumbnail_path = os.path.join(app.config['UPLOAD_FOLDER'], f'thumbnail_{clip_id}.jpg')
                    out = cv2.VideoWriter(clip_path, cv2.VideoWriter_fourcc(*'mp4v'), fps, (frame.shape[1], frame.shape[0]))
                    for f in current_clip:
                        out.write(f)
                    out.release()
                    cv2.imwrite(thumbnail_path, current_clip[0])  # Save first frame as thumbnail
                    clips.append({
                        'id': clip_id,
                        'path': clip_path,
                        'thumbnail': thumbnail_path,
                        'start_time': clip_start,
                        'duration': len(current_clip) / fps
                    })
                    thumbnails.append(thumbnail_path)
                    clip_id += 1
                current_clip = []

        frame_count += 1
        # Update progress (can be sent via WebSocket or polled)
        progress = (frame_count / total_frames) * 100
        # Simulate progress update (implement WebSocket for real-time updates)

    # Save last clip if exists
    if current_clip and len(current_clip) > fps:
        clip_path = os.path.join(app.config['UPLOAD_FOLDER'], f'clip_{clip_id}.mp4')
        thumbnail_path = os.path.join(app.config['UPLOAD_FOLDER'], f'thumbnail_{clip_id}.jpg')
        out = cv2.VideoWriter(clip_path, cv2.VideoWriter_fourcc(*'mp4v'), fps, (frame.shape[1], frame.shape[0]))
        for f in current_clip:
            out.write(f)
        out.release()
        cv2.imwrite(thumbnail_path, current_clip[0])
        clips.append({
            'id': clip_id,
            'path': clip_path,
            'thumbnail': thumbnail_path,
            'start_time': clip_start,
            'duration': len(current_clip) / fps
        })
        thumbnails.append(thumbnail_path)

    cap.release()
    return clips, thumbnails

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/upload', methods=['POST'])
def upload_video():
    if 'video' not in request.files:
        return jsonify({'error': 'No video file provided'}), 400
    file = request.files['video']
    if file and allowed_file(file.filename):
        filename = secure_filename(file.filename)
        video_path = os.path.join(app.config['UPLOAD_FOLDER'], filename)
        file.save(video_path)
        clips, thumbnails = process_video(video_path)
        return jsonify({
            'clips': clips,
            'thumbnails': thumbnails,
            'video_path': video_path
        })
    return jsonify({'error': 'Invalid file type'}), 400

@app.route('/clip/<int:clip_id>')
def serve_clip(clip_id):
    clip_path = os.path.join(app.config['UPLOAD_FOLDER'], f'clip_{clip_id}.mp4')
    if os.path.exists(clip_path):
        return send_file(clip_path, mimetype='video/mp4')
    return jsonify({'error': 'Clip not found'}), 404

if __name__ == '__main__':
    app.run(debug=True)
